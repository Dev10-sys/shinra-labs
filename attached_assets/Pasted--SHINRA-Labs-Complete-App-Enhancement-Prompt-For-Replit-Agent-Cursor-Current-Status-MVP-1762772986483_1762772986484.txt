## PART 13: DATABASE SCHEMA

```sql
-- Users Table
CREATE TABLE users (
    id INTEGER PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255),
    role ENUM('freelancer', 'company', 'admin'),
    name VARCHAR(255),
    college_or_company VARCHAR(255),
    avatar_url VARCHAR(255),
    languages JSON,
    wallet_balance DECIMAL(10, 2) DEFAULT 0,
    total_earnings DECIMAL(12, 2) DEFAULT 0,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- Tasks Table
CREATE TABLE tasks (
    id INTEGER PRIMARY KEY,
    company_id INTEGER REFERENCES users(id),
    title VARCHAR(255),
    description TEXT,
    task_type ENUM('image', 'audio', 'video', 'text'),
    difficulty ENUM('easy', 'medium', 'hard'),
    total_items INTEGER,
    items_completed INTEGER DEFAULT 0,
    reward_per_item DECIMAL(10, 2),
    total_budget DECIMAL(12, 2),
    deadline TIMESTAMP,
    status ENUM('open', 'in_progress', 'completed', 'cancelled'),
    min_accuracy FLOAT DEFAULT 90.0,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- Submissions Table
CREATE TABLE submissions (
    id INTEGER PRIMARY KEY,
    task_id INTEGER REFERENCES tasks(id),
    freelancer_id INTEGER REFERENCES users(id),
    item_data JSON,
    label_data JSON,
    accuracy FLOAT,
    status ENUM('pending', 'approved', 'rejected'),
    submitted_at TIMESTAMP,
    verified_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- Wallet/Transaction Table
CREATE TABLE transactions (
    id INTEGER PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    amount DECIMAL(10, 2),
    type ENUM('earning', 'withdrawal', 'bonus'),
    status ENUM('pending', 'completed', 'failed'),
    description VARCHAR(255),
    transaction_id VARCHAR(255),
    created_at TIMESTAMP,
    completed_at TIMESTAMP
);

-- Datasets Table
CREATE TABLE datasets (
    id INTEGER PRIMARY KEY,
    created_by INTEGER REFERENCES users(id),
    task_id INTEGER REFERENCES tasks(id),
    name VARCHAR(255),
    description TEXT,
    data_count INTEGER,
    accuracy FLOAT,
    price DECIMAL(12, 2),
    total_sales INTEGER DEFAULT 0,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- Create Indexes
CREATE INDEX idx_tasks_status ON tasks(status);
CREATE INDEX idx_tasks_company ON tasks(company_id);
CREATE INDEX idx_submissions_freelancer ON submissions(freelancer_id);
CREATE INDEX idx_submissions_task ON submissions(task_id);
CREATE INDEX idx_transactions_user ON transactions(user_id);
```
